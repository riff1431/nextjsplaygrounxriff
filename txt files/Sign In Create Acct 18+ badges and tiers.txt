                                          <span className="text-xs text-gray-200/80">{x.p}</span>
                                        </div>
                                      </button>
                                    ))}
                                  </div>
                                </div>
                              </div>

                              <div className="mt-4 rounded-2xl border border-cyan-400/20 bg-cyan-500/10 p-4">
                                <div className="text-sm font-medium text-cyan-100">What tiers do</div>
                                <ul className="mt-2 list-disc pl-5 text-xs text-cyan-100/80 space-y-1">
                                  <li>Tier badge appears on your creator cards and room headers (where enabled).</li>
                                  <li>Higher tiers get better placement in search results and featured lists.</li>
                                  <li>Helps you stand out and signals reliability to fans.</li>
                                </ul>
                              </div>
                            </div>

                            {/* Separate add-on: Suga Badge ($150/mo) */}
                            <div className="rounded-2xl border border-cyan-400/20 bg-black/35 p-4 shadow-[0_0_26px_rgba(0,230,255,0.10)]">
                              <div className="flex items-start justify-between gap-3">
                                <div>
                                  <div className="text-sm font-medium text-gray-100">Suga Badge (Suga 4 U)</div>
                                  <div className="mt-1 text-xs text-gray-200/70">
                                    Separate add-on that marks you as available in <span className="text-gray-100">Suga 4 U</span>. Flat price — not tied to the tier ladder.
                                  </div>
                                </div>
                                <Badge className="rounded-full border border-white/10 bg-black/40 text-gray-200">$150/mo</Badge>
                              </div>

                              <div className="mt-4 flex items-start gap-3 rounded-2xl border border-white/10 bg-black/30 p-4">
                                <Checkbox id="creator_suga_badge" checked={creatorSugaBadgeEnabled} onCheckedChange={(v) => setCreatorSugaBadgeEnabled(!!v)} />
                                <div className="flex-1">
                                  <Label htmlFor="creator_suga_badge" className="text-sm font-medium text-gray-100">Enable Suga Badge</Label>
                                  <div className="mt-1 text-xs text-gray-200/70">Shows a Suga badge on your profile and surfaces you in Suga 4 U filters.</div>
                                  <div className="mt-3 flex flex-wrap gap-2">
                                    <span className="rounded-full border border-white/10 bg-black/30 px-3 py-1 text-[11px] text-gray-200">Flat $150/mo</span>
                                    <span className="rounded-full border border-white/10 bg-black/30 px-3 py-1 text-[11px] text-gray-200">Independent of tiers</span>
                                    <span className="rounded-full border border-white/10 bg-black/30 px-3 py-1 text-[11px] text-gray-200">Suga 4 U discoverability</span>
                                  </div>
                                </div>
                              </div>

                              <div className="mt-3 text-xs text-gray-300/70">Preview note: Add-ons are mocked; production requires KYC + payout readiness + subscription checkout.</div>
                            </div>
                          </div>
                        )}

                        {/* STEP 3: ID Verification */}
                        {createStep === 3 && (
                          <div className="space-y-4">
                            <div className="rounded-2xl border border-pink-500/20 bg-black/35 p-4 shadow-[0_0_26px_rgba(255,0,200,0.10)]">
                              <div className="flex items-start justify-between gap-3">
                                <div>
                                  <div className="text-sm font-medium text-gray-100">ID verification (18+)</div>
                                  <div className="mt-1 text-xs text-gray-200/70">Required before you can publish or monetize. Preview-only UI.</div>
                                </div>
                                <Badge className={cx("rounded-full", verificationStatus === "verified" ? "bg-cyan-500/20 text-cyan-100 border border-cyan-400/30" : verificationStatus === "rejected" ? "bg-red-500/15 text-red-200 border border-red-500/30" : "bg-white/10 text-gray-100 border border-white/10")}>
                                  {verificationStatus === "not_started" ? "Not started" : verificationStatus === "submitted" ? "Submitted" : verificationStatus === "under_review" ? "Under review" : verificationStatus === "verified" ? "Verified" : "Rejected"}
                                </Badge>
                              </div>

                              <div className="mt-4 grid grid-cols-3 gap-2">
                                {[{ t: "Government ID", ok: idFront || idBack }, { t: "Selfie / liveness", ok: selfieDone }, { t: "Details check", ok: creatorSubmitted }].map((x) => (
                                  <div key={x.t} className="rounded-xl border border-white/10 bg-black/30 p-3">
                                    <div className="flex items-center justify-between">
                                      <div className="text-xs text-gray-200">{x.t}</div>
                                      {x.ok ? <CheckCircle2 className="h-4 w-4 text-cyan-200" /> : <Lock className="h-4 w-4 text-gray-400" />}
                                    </div>
                                  </div>
                                ))}
                              </div>
                            </div>

                            <div className="space-y-3">
                              <div className="rounded-2xl border border-white/10 bg-black/35 p-4">
                                <div className="text-sm font-medium text-gray-100">1) Upload government ID</div>
                                <div className="mt-1 text-xs text-gray-200/70">Passport, driver’s license, or national ID.</div>
                                <div className="mt-3 grid grid-cols-1 gap-3 sm:grid-cols-2">
                                  <button type="button" onClick={() => setIdFront((v) => !v)} className={cx("h-16 rounded-2xl border text-left px-4 transition", idFront ? "border-cyan-400/30 bg-cyan-500/10" : "border-white/10 bg-black/30 hover:bg-white/5")}>
                                    <div className="flex items-center justify-between">
                                      <div>
                                        <div className="text-xs text-gray-300">Front</div>
                                        <div className="text-sm text-gray-100">Tap to {idFront ? "remove" : "upload"}</div>
                                      </div>
                                      <IdCard className={cx("h-5 w-5", idFront ? "text-cyan-200" : "text-gray-400")} />
                                    </div>
                                  </button>
                                  <button type="button" onClick={() => setIdBack((v) => !v)} className={cx("h-16 rounded-2xl border text-left px-4 transition", idBack ? "border-cyan-400/30 bg-cyan-500/10" : "border-white/10 bg-black/30 hover:bg-white/5")}>
                                    <div className="flex items-center justify-between">
                                      <div>
                                        <div className="text-xs text-gray-300">Back (if applicable)</div>
                                        <div className="text-sm text-gray-100">Tap to {idBack ? "remove" : "upload"}</div>
                                      </div>
                                      <IdCard className={cx("h-5 w-5", idBack ? "text-cyan-200" : "text-gray-400")} />
                                    </div>
                                  </button>
                                </div>
                                <div className="mt-2 text-[11px] text-gray-300/70">In production, use a verified KYC provider + secure upload/tokenization flow.</div>
                              </div>

                              <div className="rounded-2xl border border-white/10 bg-black/35 p-4">
                                <div className="text-sm font-medium text-gray-100">2) Selfie / liveness</div>
                                <div className="mt-1 text-xs text-gray-200/70">Match your selfie with the ID photo.</div>
                                <div className="mt-3 flex items-center gap-3">
                                  <div className={cx("h-14 w-14 rounded-full border", selfieDone ? "border-cyan-400/30 bg-cyan-500/10" : "border-white/10 bg-black/30")}>
                                    <div className="flex h-full w-full items-center justify-center">
                                      <Camera className={cx("h-5 w-5", selfieDone ? "text-cyan-200" : "text-gray-400")} />
                                    </div>
                                  </div>
                                  <Button type="button" className="h-11 rounded-xl bg-pink-600 hover:bg-pink-700" onClick={() => setSelfieDone(true)} disabled={selfieDone}>
                                    {selfieDone ? "Selfie complete" : "Start selfie check"}
                                  </Button>
                                  {selfieDone && <Badge className="rounded-full bg-cyan-500/20 text-cyan-100 border border-cyan-400/30">Liveness passed</Badge>}
                                </div>
                              </div>

                              <div className="rounded-2xl border border-white/10 bg-black/35 p-4">
                                <div className="text-sm font-medium text-gray-100">3) Legal attestations</div>
                                <div className="mt-3 space-y-3">
                                  <div className="flex items-start gap-2">
                                    <Checkbox id="attest18" checked={attest18} onCheckedChange={(v) => setAttest18(!!v)} />
                                    <Label htmlFor="attest18" className="text-sm font-normal leading-relaxed text-gray-200/80">I confirm I am 18+.</Label>
                                  </div>
                                  <div className="flex items-start gap-2">
                                    <Checkbox id="attestSelf" checked={attestSelf} onCheckedChange={(v) => setAttestSelf(!!v)} />
                                    <Label htmlFor="attestSelf" className="text-sm font-normal leading-relaxed text-gray-200/80">I am the person shown on this ID.</Label>
                                  </div>
                                  <div className="flex items-start gap-2">
                                    <Checkbox id="attestConsent" checked={attestConsent} onCheckedChange={(v) => setAttestConsent(!!v)} />
                                    <Label htmlFor="attestConsent" className="text-sm font-normal leading-relaxed text-gray-200/80">I consent to identity verification processing.</Label>
                                  </div>
                                </div>
                              </div>

                              {verificationStatus === "rejected" && (
                                <div className="rounded-2xl border border-red-500/30 bg-red-500/10 p-4">
                                  <div className="flex items-center gap-2 text-sm font-medium text-red-200"><XCircle className="h-4 w-4" /> Verification rejected</div>
                                  <div className="mt-2 text-xs text-red-100/80">Reason: blurry image / mismatch (mock). Please re-submit.</div>
                                </div>
                              )}

                              <div className="flex gap-2">
                                <Button variant="outline" className="h-11 w-full rounded-xl border-white/10 bg-black/35 text-gray-100 hover:bg-white/5" onClick={prevCreateStep}>
                                  Back
                                </Button>
                                <Button
                                  className="h-11 w-full rounded-xl bg-pink-600 hover:bg-pink-700"
                                  disabled={!creatorReadyToSubmit || creatorSubmitted}
                                  onClick={() => {
                                    setVerificationStatus("submitted");
                                    // Simulate review progression in preview
                                    setTimeout(() => setVerificationStatus("under_review"), 600);
                                    setTimeout(() => setVerificationStatus("verified"), 1400);
                                  }}
                                >
                                  {creatorSubmitted ? (
                                    <span className="inline-flex items-center gap-2"><Loader2 className="h-4 w-4 animate-spin" /> Submitted</span>
                                  ) : (
                                    "Submit for verification"
                                  )}
                                </Button>
                              </div>

                              <Button
                                className="h-11 w-full rounded-xl"
                                variant={creatorVerified ? "default" : "outline"}
                                onClick={() => creatorVerified && nextCreateStep()}
                                disabled={!creatorVerified}
                              >
                                {creatorVerified ? "Continue" : "Continue (locked until verified)"}
                              </Button>

                              <div className="text-xs text-gray-300/70">
                                Security note (preview): In production, verification should be handled by a certified identity provider using redirect/tokenized uploads.
                              </div>
                            </div>
                          </div>
                        )}

                        {/* STEP 4: Payouts */}
                        {createStep === 4 && (
                          <div className="space-y-4">
                            <div className="rounded-2xl border border-white/10 bg-black/35 p-4">
                              <div className="flex items-center justify-between">
                                <div>
                                  <div className="text-sm font-medium text-gray-100">Payouts (preview)</div>
                                  <div className="mt-1 text-xs text-gray-200/70">Connect payouts to receive earnings. Monetization remains gated by verification.</div>
                                </div>
                                <CreditCard className="h-5 w-5 text-cyan-200" />
                              </div>
                            </div>

                            <div className="rounded-2xl border border-white/10 bg-black/35 p-4">
                              <div className="text-sm font-medium text-gray-100">Choose a payout method</div>
                              <div className="mt-3 grid grid-cols-1 gap-2">
                                {[
                                  { t: "Bank transfer", d: "Standard payouts (2–5 days)" },
                                  { t: "Card payout", d: "Fast payouts (fees may apply)" },
                                  { t: "Provider", d: "Connect via a payout partner" },
                                ].map((m) => (
                                  <button key={m.t} type="button" className="rounded-2xl border border-white/10 bg-black/30 p-4 text-left hover:bg-white/5">
                                    <div className="text-sm font-medium text-gray-100">{m.t}</div>
                                    <div className="text-xs text-gray-200/70">{m.d}</div>
                                  </button>
                                ))}
                              </div>
                            </div>

                            <div className="flex gap-2">
                              <Button variant="outline" className="h-11 w-full rounded-xl border-white/10 bg-black/35 text-gray-100 hover:bg-white/5" onClick={prevCreateStep}>
                                Back
                              </Button>
                              <Button className="h-11 w-full rounded-xl bg-pink-600 hover:bg-pink-700">
                                Finish setup
                              </Button>
                            </div>

                            <div className="rounded-2xl border border-cyan-400/20 bg-cyan-500/10 p-4">
                              <div className="flex items-center gap-2 text-sm font-medium text-cyan-100"><CheckCircle2 className="h-4 w-4" /> Creator Mode: {creatorVerified ? "Active" : "Pending"}</div>
                              <div className="mt-2 text-xs text-cyan-100/80">Publishing + monetization unlock only when verified.</div>
                            </div>
                          </div>
                        )}
                      </>
                    )}

                    {/* Social (always available for Create) */}
                    <div className="relative">
                      <Separator className="bg-white/10" />
                      <span className="absolute left-1/2 top-1/2 -translate-x-1/2 -translate-y-1/2 rounded-full bg-black/60 px-3 text-xs text-gray-300 border border-white/10">
                        or continue with
                      </span>
                    </div>

                    <div className="grid grid-cols-2 gap-3">
                      <SocialButton>
                        <Sparkles className="h-4 w-4" /> Google
                      </SocialButton>
                      <SocialButton>
                        <Crown className="h-4 w-4" /> Apple
                      </SocialButton>
                    </div>
                  </TabsContent>
                </Tabs>

                <div className="text-center text-xs text-gray-300/70">
                  Preview-only UI. Production auth should include risk checks, device verification, and age gating.
                </div>
              </CardContent>
            </Card>
          </motion.div>
        </div>
      </div>
    </div>
  );
}

// components/tours/FlashDropFanTour.jsx

"use client";

import React, { useEffect, useState } from "react";
import Joyride, { STATUS } from "react-joyride";

export default function FlashDropFanTour({ user }) {
  const [runTour, setRunTour] = useState(false);

  const steps = [
    {
      target: ".flashdrop-fan-creator-info",
      title: "1. Creator Info",
      content:
        "View the creator's profile picture and connection status.",
      placement: "right",
    },
    {
      target: ".flashdrop-fan-quick-actions",
      title: "2. Quick Actions",
      content:
        "Use these buttons to follow or add the creator, share the room, report any issues, and view incoming content sent from the creator.",
      placement: "right",
    },
    {
      target: ".flashdrop-fan-live-drop-board",
      title: "3. Live Drop Board",
      content:
        "See current drops available from the creator. Tap on Photos or Videos to browse content.",
      placement: "right",
    },
    {
      target: ".flashdrop-fan-live-chat",
      title: "4. Live Chat",
      content:
        "Chat in real-time with the creator and other fans in the room.",
      placement: "left",
    },
    {
      target: ".flashdrop-fan-reactions",
      title: "5. Reactions",
      content:
        "Send fun reactions to the creator and stand out in the chat. More reactions, more love!",
      placement: "left",
    },
    {
      target: ".flashdrop-fan-high-roller-packs",
      title: "6. High Roller Packs",
      content:
        "Check out exclusive premium packs created by the creator. Exclusive content inside!",
      placement: "left",
    },
    {
      target: ".flashdrop-fan-request-drop",
      title: "7. Request a Drop",
      content:
        "Request custom photos, videos, or special content from the creator. Be clear and add your offer.",
      placement: "left",
    },
    {
      target: ".flashdrop-fan-top-alerts",
      title: "8. Top Alerts",
      content:
        "Stay updated with important announcements, rare items, limited drops, and special events.",
      placement: "bottom",
    },
  ];

  useEffect(() => {
    if (!user?.flashDropFanTourCompleted) {
      const timer = setTimeout(() => {
        setRunTour(true);
      }, 700);

      return () => clearTimeout(timer);
    }
  }, [user]);

  const handleTourCallback = async (data) => {
    const finishedStatuses = [STATUS.FINISHED, STATUS.SKIPPED];

    if (finishedStatuses.includes(data.status)) {
      setRunTour(false);

      await fetch("/api/user/complete-room-tour", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
        },
        body: JSON.stringify({
          userId: user.id,
          tourName: "flashdrop_fan",
        }),
      });
    }
  };

  return (
    <Joyride
      steps={steps}
      run={runTour}
      continuous
      showSkipButton
      showProgress
      disableOverlayClose
      spotlightClicks
      callback={handleTourCallback}
      locale={{
        back: "Back",
        close: "Close",
        last: "Finish Tour",
        next: "Next",
        skip: "Skip Tour",
      }}
      styles={{
        options: {
          zIndex: 999999,
          backgroundColor: "#120814",
          textColor: "#ffffff",
          primaryColor: "#ff2fb3",
          arrowColor: "#120814",
          overlayColor: "rgba(0, 0, 0, 0.78)",
          spotlightShadow: "0 0 35px rgba(255, 47, 179, 0.9)",
        },
        tooltip: {
          borderRadius: "18px",
          padding: "18px",
          border: "1px solid rgba(255, 47, 179, 0.7)",
          boxShadow:
            "0 0 30px rgba(255, 47, 179, 0.4), 0 0 55px rgba(255, 193, 7, 0.18)",
          background:
            "linear-gradient(135deg, rgba(18,8,20,0.98), rgba(40,10,35,0.98))",
        },
        tooltipTitle: {
          color: "#ff4fd8",
          fontSize: "18px",
          fontWeight: "800",
        },
        tooltipContent: {
          color: "#ffffff",
          fontSize: "14px",
          lineHeight: "1.6",
        },
        buttonNext: {
          background: "linear-gradient(90deg, #ff2fb3, #ffb000)",
          borderRadius: "999px",
          padding: "10px 18px",
          fontWeight: "800",
          color: "#ffffff",
        },
        buttonBack: {
          color: "#ffffff",
        },
        buttonSkip: {
          color: "#cfc2d8",
        },
      }}
    />
  );
}
